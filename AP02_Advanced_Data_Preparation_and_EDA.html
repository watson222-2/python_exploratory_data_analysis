price = []

    d1 = re.findall("CHF\s*(\d+)\.-", i)
    try:
        ### strip() removes leading and trailing whitespaces, 
        ### d2 = d1[0].strip().replace(',', '.')
        d2 = d1[0].strip()
    except:
        d2 = None
    price.append(d2)
